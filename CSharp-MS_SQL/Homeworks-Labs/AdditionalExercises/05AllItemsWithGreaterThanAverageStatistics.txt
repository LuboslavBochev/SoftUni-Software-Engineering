DECLARE @avgSpeed INT = (SELECT AVG(s.Speed) FROM Items i INNER JOIN [Statistics] s ON s.Id = i.StatisticId)
DECLARE @avgMind INT = (SELECT AVG(s.Mind) FROM Items i INNER JOIN [Statistics] s ON s.Id = i.StatisticId)
DECLARE @avgLuck INT = (SELECT AVG(s.Luck) FROM Items i INNER JOIN [Statistics] s ON s.Id = i.StatisticId)

SELECT i.Name,i.Price,i.MinLevel, s.Strength, s.Defence,s.Speed AS [Speed], s.Luck AS [Luck], s.Mind AS [Mind] FROM Items AS i
	INNER JOIN [Statistics] AS s ON s.Id = i.StatisticId
		WHERE Speed > @avgSpeed AND Luck > @avgLuck AND Mind > @avgMind
		ORDER BY i.Name
